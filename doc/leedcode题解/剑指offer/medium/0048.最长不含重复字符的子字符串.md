## 题目描述

```cpp
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        if(s.size() == 0)return 0;
        
        int n  = s.size();
        unordered_map<char, int> hashmap;
        int start = 0, end = 0;
        int maxLength = 0;
        int curLength = 0;
        while(end < n)
        {
            if(hashmap.count(s[end]) && hashmap[s[end]] >= start)
            {     
                start = hashmap[s[end]] + 1;
                curLength = end - start;             
            }
            curLength++;
            hashmap[s[end]] = end;
            end++;
            maxLength = max(maxLength, curLength);
        }
        return maxLength;
    }
};
```
